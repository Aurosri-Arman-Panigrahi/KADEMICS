<script type="module">
import { requireAuth, initProfileUI } from './js/auth.js';
import { API } from './js/api.js';

const user = await requireAuth();

initProfileUI(user);

const params   = new URLSearchParams(location.search);
const teacherId = parseInt(params.get('id'));
let selectedRating = 0;
let allSections = [];
let activeSectionId = null;

window.setRating      = setRating;
window.submitRating   = submitRating;
window.submitComment  = submitComment;
window.loadSection    = loadSection;

async function loadDossier() {
  const data = await API.getTeacherDossier(teacherId);
  document.title = `KADEMICS  ${data.name}`;
  document.getElementById('t-name').textContent = data.name || '';
  document.getElementById('t-qual').textContent = data.qualification || '';
  document.getElementById('t-bio').textContent  = data.bio || '';

  const avg = data.avg_rating || 0;
  document.getElementById('t-avg').textContent = avg.toFixed(1);
  document.getElementById('t-diamonds').innerHTML = diamonds(avg);

  if (data.photo) {
    const w = document.getElementById('t-photo-wrap');
    w.innerHTML = '';
    const img = document.createElement('img');
    img.src = data.photo; img.alt = data.name;
    img.style.cssText='width:140px;height:140px;border-radius:50%;object-fit:cover;';
    w.appendChild(img);
  }

  // Sections
  allSections = data.sections || [];
  renderSectionPills();
  if (allSections.length) loadSection(allSections[0].id);
}

function renderSectionPills() {
  const pArea = document.getElementById('section-pills');
  const sel   = document.getElementById('comment-section-select');
  pArea.innerHTML = ''; sel.innerHTML = '';
  allSections.forEach(s => {
    const pill = document.createElement('span');
    pill.className = 'section-pill';
    pill.textContent = s.section_name || s.name;
    pill.onclick = () => loadSection(s.id, pill);
    pArea.appendChild(pill);

    const opt  = document.createElement('option');
    opt.value  = s.id; opt.textContent = s.section_name || s.name;
    sel.appendChild(opt);
  });
  if (!allSections.length) {
    pArea.innerHTML = '<p class="text-dim" style="font-size:0.8rem">No sections added yet.</p>';
  }
}

async function loadSection(sectionId, pillEl=null) {
  activeSectionId = sectionId;
  document.querySelectorAll('.section-pill').forEach(p => p.classList.remove('active'));
  if (pillEl) pillEl.classList.add('active');

  const comments = await API.getComments(sectionId);
  const area = document.getElementById('comments-area');
  area.innerHTML = '';
  (comments || []).forEach(c => {
    const div = document.createElement('div');
    div.className = 'comment-item animate-fade-in';
    div.innerHTML = `
      <p class="comment-meta"><i class="fa fa-user" style="color:var(--cyan)"></i> ${escHtml(c.commenter_name||c.roll)} &nbsp;|&nbsp; ${c.created_at?.substring(0,10)||''}</p>
      <p style="font-size:0.85rem;line-height:1.6">${escHtml(c.message||c.content||'')}</p>`;
    area.appendChild(div);
  });
  if (!area.children.length) {
    area.innerHTML = '<p class="text-dim" style="font-size:0.8rem;padding:var(--space-md)">No reviews yet. Be the first!</p>';
  }
}

function setRating(v) {
  selectedRating = v;
  document.querySelectorAll('.star-pick').forEach(s => {
    s.classList.toggle('lit', parseInt(s.dataset.v) <= v);
  });
  document.getElementById('btn-rate').disabled = false;
}

async function submitRating() {
  if (!selectedRating) return;
  try {
    const r2 = await API.rateTeacher({ teacher_id:teacherId, roll:user.roll, score:selectedRating, section_id:activeSectionId||0 });
    document.getElementById('t-avg').textContent = r2.new_avg?.toFixed(1) || '';
    document.getElementById('t-diamonds').innerHTML = diamonds(r2.new_avg||0);
    showToast('success', `Rating submitted: ${selectedRating} ?`);
  } catch(e) { showToast('error','Failed to submit rating'); }
}

async function submitComment() {
  const sectionId = document.getElementById('comment-section-select')?.value;
  const msg       = document.getElementById('comment-text')?.value.trim();
  if (!msg) { showToast('error','Review cannot be empty'); return; }
  try {
    await API.addComment({ section_id:sectionId, roll:user.roll, message:msg });
    document.getElementById('comment-text').value = '';
    showToast('success','Review transmitted!');
    if (sectionId) loadSection(sectionId);
  } catch(e) { showToast('error','Submission failed'); }
}

function diamonds(avg) {
  let h = '';
  for (let i=1;i<=5;i++) h += `<span style="color:${i<=Math.round(avg)?'var(--cyan)':'var(--text-dim)'};font-size:1.2rem">?</span>`;
  return h;
}
function showToast(type, msg) {
  const c = document.getElementById('toast-container');
  const t = document.createElement('div');
  t.className = `toast toast-${type}`;
  t.innerHTML = `<i class="fa fa-${type==='success'?'check':type==='error'?'times':'info-circle'}"></i> ${msg}`;
  c.appendChild(t); setTimeout(() => t.remove(), 4000);
}
function escHtml(s){ return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }

if (teacherId) loadDossier();
else { document.getElementById('t-name').textContent = 'DOSSIER NOT FOUND'; }
</script>
